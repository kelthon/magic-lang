program         := 'class' '{' declarations '}';

declarations    := declaration declarations
                 | Îµ ;

declaration     := type identifier ('=' expression)? ';' ;

expression      := assignment ;

assignment      := identifier '=' assignment
                 | equality ;

equality        := comparison equality_binary ;
equality_binary := (('==' | '!=') comparison)* ;

comparison      := term comparison_binary ;
comparison_binary := (('>=' | '<=' | '>' | '<') term)* ;

term            := factor term_binary ;
term_binary     := (('+' | '-') factor)* ;

factor          := unary factor_binary ;
factor_binary   := (('*' | '/') unary)* ;

unary           := primary ;

primary         := literal
                 | identifier
                 | '(' expression ')' ;

literal         := float_literal
                 | integer_literal
                 | char_literal ;

type            := 'int'
                 | 'string'
                 | 'bool'
                 | 'float'
                 | 'double'
                 | 'char'
                 | identifier ;

char_literal    := '\'' ('\\'? character)? '\'' ;

float_literal   := integer fractional_part ;

integer_literal := digits ;

fractional_part := '.' digits ;

digits          := [0-9]+ ;

identifier      := [a-zA-Z_][a-zA-Z0-9_]* ;
